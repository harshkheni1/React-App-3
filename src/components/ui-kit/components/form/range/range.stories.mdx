# Range Examples
<Preview>
  <Form
    defaultValues={{
      testRange: {
        min: 15,
        max: 150,
      },
    }}
  >
    {api => (
      <>
        <FormLabel>Base price / package:</FormLabel>
        <Box mb={15}>
          <Row>
            <Col col={6}>
              <FormInput type="number" name="testRange.min" />
            </Col>
            <Col col={6}>
              <FormInput type="number" name="testRange.max" />
            </Col>
          </Row>
        </Box>
        <FormRange
          min={0}
          max={200}
          defaultValue={[api.getValue('testRange.min'), api.getValue('testRange.max')]}
          value={[api.getValue('testRange.min'), api.getValue('testRange.max')]}
          step={1}
          onChange={value => {
            api.setValue('testRange.min', value[0]);
            api.setValue('testRange.max', value[1]);
          }}
        />
        <Button type="submit">Submit</Button>
      </>
    )}
  </Form>
</Preview>
